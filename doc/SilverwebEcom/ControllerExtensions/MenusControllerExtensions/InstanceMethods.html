def update_checkbox_tag
  @menu = Menu.find(params[:id])
  @tag_name = params[:tag_name] || 'tag_list'

  if @menu.send(@tag_name).include?(params[:field])
    @menu.send(@tag_name).remove(params[:field])
  else
    @menu.send(@tag_name).add(params[:field])
  end

  @menu.save

  head :ok
end
