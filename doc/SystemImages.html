def self.method_missing(method, *args)
  if respond_to?(method)
    super
  else
    method_name = method.to_s

    # set a value for a variable
    if /=$/.match?(method_name)
      var_name = method_name.delete('=')
      value = args.first
      self[var_name] = value

      # retrieve a value
    else
      self[method_name]

    end
  end
