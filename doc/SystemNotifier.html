def display_page_body(page_name)
  page = begin
    Page.where(title: page_name).first
  rescue StandardError
    ''
  end

  begin
    return page.body.html_safe
  rescue StandardError
    "Page '#{page_name}' not found.   Please create it in pages."
  end
end
